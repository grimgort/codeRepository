417 bool matchesCategories (SceneNode::Pair&
418  colliders, Category::Type type1,
419  Category::Type type2)
420 {
421  unsigned int category1 =
422  colliders.first->getCategory();
423  unsigned int category2 =
424  colliders.second->getCategory();
425 
426  if (type1 & category1 && type2 & category2)
427  {
428  return true;
429  }
430  else if (type1 & category2 && type2 & category2)

432  return true;
