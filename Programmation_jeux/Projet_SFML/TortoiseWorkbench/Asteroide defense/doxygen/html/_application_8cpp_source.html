67 void Application::run()
68 {
69  sf::Clock clock;
70  sf::Time timeSinceLastUpdate =
71  sf::Time::Zero; // define time of each frame (initialisé à 0)
72 
73  // RQ : If Application lag, you can take it smoother with called multi time. Look lesson on it(p24).
74  // You can activate V-Sync for adapts time step with your monitory(p24)
75  while (m_window.isOpen()) //On fait tourner l'application tant que la fenêtre principal est ouverte.
76  {
77  sf::Time dt = clock.restart(); //on remet à 0 le pas de temps
78  timeSinceLastUpdate += dt;
79 
80  while (timeSinceLastUpdate > timePerFrame)
81  {
82  timeSinceLastUpdate -= timePerFrame ;
83  //Appel la pile des événements
84  processInput();
85  //Met à jour l'ensemble des calcul du jeux en fonction du dt entre 2 frame
86  update (timePerFrame);
87 
88  //Ferme l'application si le stack des fenêtre est vide
89  if (m_stateStack.isEmpty())
90  { m_window.close(); }
91  }
92 
93  //Ecrit le nombre de frame maximum(performence du jeux)
94  updateStatistics (dt);
95  //Trace les texture
96  render();
97  }
98 }
